% Page Numbering for long reports
\makeatletter
\newcommand\frontmatter{%
    \cleardoublepage
    %\@mainmatterfalse
    \pagenumbering{roman}}

\newcommand\mainmatter{%
    \cleardoublepage
    % \@mainmattertrue
    \pagenumbering{arabic}}

\newcommand\backmatter{%
    \if@openright
        \cleardoublepage
    \else
        \clearpage
    \fi
    % \@mainmatterfalse
}
\makeatother
